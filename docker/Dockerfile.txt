FROM python:3.10-slim

# Install system dependencies required by the TTS script
RUN apt-get update && apt-get install -y espeak-ng ffmpeg && rm -rf /var/lib/apt/lists/*

# Set up the application directory
WORKDIR /app
COPY relay_example.py .

# Expose the hardcoded port from the script
EXPOSE 5005

# Run the script
CMD ["python3", "relay_example.py"]
